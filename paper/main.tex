\begin{abstract}
We reformulate the Projection Spectral Layer Theory (PSLT) as a first-principles projection-spectral derivation of the three-generation Standard Model.
The central geometric input is no longer a scan-level three-layer occupancy pattern, but a two-center projection divisor
\(R=p_+ + p_-\) on the compact projection surface \(C=\CP^1\).
The generation bundle is
\(L_R=\cO(R)\simeq\cO(2)\).
By Riemann--Roch and Serre duality,
\[
\dim H^0(\CP^1,\cO(2))=3,
\]
so the low-energy chiral Hilbert space contains exactly three copies of the anomaly-free Standard Model family representation and no fourth chiral generation.
The same two-center geometry yields the conformal factor \(\Omega\) through a variational Poisson constraint and induces the PSLT spectral operator
\([-\nabla^2+V_{\rm eff}]\psi_N=\omega_N^2\psi_N\).
The former closure quantities \(g_N,\Gamma_N,B_N\) are redefined as spectral degeneracy, two-center Schur-complement kinetics, and Weyl-normalized visible-sector overlap, respectively.
The normalized probability
\[
P_N(t)=
\frac{B_Ng_N(1-\ee^{-\Gamma_Nt})}{\sum_K B_Kg_K(1-\ee^{-\Gamma_Kt})}
\]
then follows as a conditional spectral-measure probability generated by the formation semigroup.
Thus the three-generation structure is a topological consequence of the two-center projection geometry rather than a numerical scan artifact.
\end{abstract}

\section{Core Claim}
\label{sec:core_claim}

The rewritten PSLT statement is the following:
\begin{equation}
\boxed{
\cH_{\rm chiral}^{\rm 4D}
=
\cR_{\rm SM}^{(1)}
\otimes
H^0\!\left(\CP^1,\cO(p_+ + p_-)\right).
}
\label{eq:core_hilbert}
\end{equation}
Since \(p_+\) and \(p_-\) are the two projection branch points,
\begin{equation}
\cO(p_+ + p_-)\simeq \cO(2),
\qquad
\dim H^0(\CP^1,\cO(2))=3.
\label{eq:core_dim}
\end{equation}
Therefore
\begin{equation}
\boxed{
\cH_{\rm chiral}^{\rm 4D}
=
\cR_{\rm SM}^{(1)}
\oplus
\cR_{\rm SM}^{(2)}
\oplus
\cR_{\rm SM}^{(3)}.
}
\label{eq:three_sm_families}
\end{equation}
This is the main replacement of the earlier closure logic.
The probability law \(P_N\) remains useful for formation and visibility phenomenology, but it is no longer the origin of the family count.
The family count is fixed before any numerical scan, by the zero-mode dimension of the projection line bundle.

\begin{remark}

Changing \(C\), adding projection branch points, or replacing \(L_R\) changes the topological sector and is a different theory.

\begin{equation}
C\simeq \CP^1,

R=p_+ + p_-.

L_R=\cO(R)\simeq\cO(2).
\label{eq:LR_def}

S_{\rm FP}
=
S_{\rm grav}

+S_{\rm fermion}
+S_{\Omega}
+S_{\rm proj}
+S_{\rm vis}.
\label{eq:parent_action}

G_{\rm SM}=SU(3)_c\times SU(2)_L\times U(1)_Y,

Equation~\eqref{eq:Sproj} is a compact way of saying that the two centers define the divisor of the generation bundle itself, not merely a scan ansatz.

\begin{equation}
\delta S_\Omega
=

\end{equation}
Thus
\begin{equation}
\boxed{
\nabla^2\Omega(x)=-4\pi\sigma(x).

\label{eq:Omega_poisson}
\end{equation}
For the regularized Coulomb kernel

\rho_\varepsilon(r)
=

\begin{equation}

\Omega(\rho,z;D)
=
1
+a\left(\frac{1}{r_+}+\frac{1}{r_-}\right),

r_\pm
=
\sqrt{\rho^2+(z\mp D/2)^2+\varepsilon^2}.
}
\label{eq:Omega_solution}
\end{equation}
This promotes the old projected-Poisson background to an action-derived constraint of the parent theory.

\section{One Standard Model Family from Anomaly Cancellation}
\label{sec:one_family}

The topological argument fixes the number of copies.
We first recall why the minimal one-family hypercharge pattern is fixed, up to overall normalization, by gauge invariance and anomaly cancellation.

Take the left-handed Weyl fields
\begin{equation}
Q:(3,2)_{Y_Q},\quad
u^c:(\bar 3,1)_{Y_u},\quad
d^c:(\bar 3,1)_{Y_d},
\qquad
L:(1,2)_{Y_L},\quad
e^c:(1,1)_{Y_e},
\end{equation}
and a Higgs doublet
\begin{equation}
H:(1,2)_h.
\end{equation}
Set \(Y_Q=q\).
Yukawa gauge invariance requires
\begin{align}
QHu^c:&\qquad q+h+Y_u=0,
&
QH^\dagger d^c:&\qquad q-h+Y_d=0,
\\
LH^\dagger e^c:&\qquad Y_L-h+Y_e=0.
\end{align}
Hence
\begin{equation}
Y_u=-q-h,\qquad
Y_d=-q+h,\qquad
Y_e=h-Y_L.
\label{eq:yukawa_hypercharges}
\end{equation}
The \(SU(3)^2U(1)\) anomaly gives
\begin{equation}
2Y_Q+Y_u+Y_d=0,
\end{equation}
which is automatically satisfied by Eq.~\eqref{eq:yukawa_hypercharges}.
The \(SU(2)^2U(1)\) anomaly gives
\begin{equation}
3Y_Q+Y_L=0,
\qquad
Y_L=-3q.
\label{eq:su2_anomaly}
\end{equation}
The mixed gravitational anomaly gives
\begin{equation}
6Y_Q+3Y_u+3Y_d+2Y_L+Y_e=0.
\label{eq:grav_anomaly}
\end{equation}
Substituting Eqs.~\eqref{eq:yukawa_hypercharges} and \eqref{eq:su2_anomaly},
\begin{equation}
6q+3(-q-h)+3(-q+h)+2(-3q)+(h+3q)=0,
\end{equation}
so
\begin{equation}
h=3q.
\end{equation}
Therefore
\begin{equation}
Y_u=-4q,\qquad
Y_d=2q,\qquad
Y_L=-3q,\qquad
Y_e=6q,\qquad
Y_H=3q.
\label{eq:hypercharge_solution_q}
\end{equation}
The \(U(1)^3\) anomaly then cancels:
\begin{equation}
6q^3+3(-4q)^3+3(2q)^3+2(-3q)^3+(6q)^3
=
(6-192+24-54+216)q^3=0.
\end{equation}
With the conventional normalization \(q=1/6\),
\begin{equation}
\boxed{
(Y_Q,Y_u,Y_d,Y_L,Y_e,Y_H)
=
\left(
\frac16,
-\frac23,
\frac13,
-\frac12,
1,
\frac12
\right).
}
\label{eq:SM_hypercharges}
\end{equation}
Adding a sterile \(\nu^c:(1,1)_0\) does not affect these anomaly equations.

\begin{proposition}
For the minimal chiral SM field content with the Yukawa couplings above, the one-family hypercharge pattern is fixed up to overall \(U(1)_Y\) normalization.
\end{proposition}

The number of such anomaly-free copies is not fixed by anomaly cancellation.
That number is fixed by the projection geometry in the next section.

\section{The Three-Generation Theorem}
\label{sec:three_generation_theorem}

\begin{theorem}[Two-center projection implies three generations]
\label{thm:three_gen}
Let \(C=\CP^1\), let \(R=p_+ + p_-\), and let \(L_R=\cO(R)\).
Then
\begin{equation}
\dim H^0(C,L_R)=3.
\end{equation}
Consequently, the four-dimensional low-energy chiral SM spectrum contains exactly three generations:
\begin{equation}
\cH_{\rm chiral}^{\rm 4D}
=
\cR_{\rm SM}^{(1)}
\oplus
\cR_{\rm SM}^{(2)}
\oplus
\cR_{\rm SM}^{(3)}.
\end{equation}
\end{theorem}

\begin{proof}
Riemann--Roch for a line bundle \(L\) on a compact Riemann surface \(C\) gives
\begin{equation}
h^0(C,L)-h^1(C,L)=\deg L+1-g(C),
\label{eq:RR}
\end{equation}
where \(h^i(C,L)=\dim H^i(C,L)\).
For \(C=\CP^1\), \(g(C)=0\).
Because \(R=p_+ + p_-\), \(\deg L_R=2\).
Thus
\begin{equation}
h^0(C,L_R)-h^1(C,L_R)=2+1=3.
\label{eq:RR_for_LR}
\end{equation}
By Serre duality,
\begin{equation}
h^1(C,L_R)=h^0(C,K_C\otimes L_R^{-1}).
\end{equation}
On \(\CP^1\), \(K_C=\cO(-2)\), while \(L_R^{-1}=\cO(-2)\).
Hence
\begin{equation}
K_C\otimes L_R^{-1}\simeq\cO(-4).
\end{equation}
A negative-degree line bundle on \(\CP^1\) has no nonzero holomorphic sections, so
\begin{equation}
h^0(\CP^1,\cO(-4))=0,
\qquad
h^1(C,L_R)=0.
\end{equation}
Equation~\eqref{eq:RR_for_LR} therefore gives
\begin{equation}
h^0(C,L_R)=3.
\end{equation}
Tensoring these three zero modes with the one-family SM representation gives exactly three chiral SM families.
\end{proof}

Equivalently, in index-theorem form,
\begin{equation}
\operatorname{ind}\bar\partial_{L_R}
=
\int_C \operatorname{ch}(L_R)\operatorname{td}(T_C)
=
\deg L_R+1-g(C)
=3.
\label{eq:index_form}
\end{equation}
The vanishing of \(h^1\) upgrades the index to the actual zero-mode count.

\subsection{Explicit zero-mode basis}

Choose an affine coordinate \(z\) on \(\CP^1\).
A basis of \(H^0(\CP^1,\cO(2))\) is
\begin{equation}
s_0(z)=1,\qquad
s_1(z)=z,\qquad
s_2(z)=z^2.
\label{eq:monomial_basis}
\end{equation}
Thus the chiral field expansion is
\begin{equation}
\Psi_\alpha(x,z)
=
\sum_{I=0}^{2}
\psi_{\alpha I}(x)s_I(z)
+\sum_{\lambda>0}
\Psi_{\alpha,\lambda}(x)\chi_\lambda(z).
\label{eq:fermion_expansion}
\end{equation}
The first sum is the protected chiral sector.
The second sum contains massive nonchiral modes in
\begin{equation}
\Ker(\bar\partial_{L_R})^\perp.
\end{equation}
They may exist as heavy excitations, but they are not additional SM generations.

\subsection{Why there is no fourth generation}
\label{sec:no_fourth}

A fourth chiral SM generation would require a fourth independent holomorphic section
\begin{equation}
s_3\in H^0(\CP^1,\cO(2)),
\qquad
s_3\notin \Span\{1,z,z^2\}.
\end{equation}
But Theorem~\ref{thm:three_gen} proves that this vector space has dimension exactly three.
Therefore no fourth chiral SM generation exists in this topological sector.
The only ways to obtain four chiral zero modes are to change the theory, for example by replacing \(R=p_+ + p_-\) with a degree-three divisor or by changing the compact projection surface.

\subsection{Exact numerical sanity check}
\label{sec:numerical_check}

The proof is topological, but the dimension count has a simple finite-dimensional check.
Evaluate the three basis sections at \(z=-1,0,1\):
\begin{equation}
V=
\begin{pmatrix}
1&-1&1\\
1&0&0\\
1&1&1
\end{pmatrix}.
\end{equation}
Then
\begin{equation}
\det V=2\neq0,
\qquad
\rank V=3.
\end{equation}
A direct floating-point singular-value check gives
\begin{equation}
\sigma(V)\simeq(2.13577921,\;1.41421356,\;0.66215345),
\end{equation}
again confirming that the three holomorphic sections are independent.
No fourth independent section can be tested because \(\cO(2)\) admits only quadratic homogeneous polynomials.
This numerical check is not a substitute for Riemann--Roch; it is a transparent verification of the explicit basis in Eq.~\eqref{eq:monomial_basis}.

\section{Geometry-to-Spectrum Map}
\label{sec:geometry_spectrum}

The three-generation theorem is topological and does not depend on the scalar scan.
The scalar spectral chain remains important because it supplies the PSLT formation rates and visible overlap profiles.

Take a locally conformally flat metric
\begin{equation}
g_{\mu\nu}=\Omega^2(x)\eta_{\mu\nu}.
\end{equation}
For a curvature-coupled scalar,
\begin{equation}
(\Box_g-m_0^2-\xi R)\Phi=0.
\end{equation}
Set
\begin{equation}
\Phi(t,x)=\ee^{-\ii\omega t}\Omega^{-1}(x)\psi(x).
\end{equation}
Using the conformal identity
\begin{equation}
\left(\Box_g-\frac16R\right)(\Omega^{-1}\psi)
=
\Omega^{-3}\Box_\eta\psi
\end{equation}
and
\begin{equation}
R=-6\Omega^{-3}\nabla^2\Omega,
\end{equation}
one obtains
\begin{equation}
\left[
-\nabla^2
+m_0^2\Omega^2
+(1-6\xi)\Omega^{-1}\nabla^2\Omega
\right]\psi
=
\omega^2\psi.
\end{equation}
Define
\begin{equation}
V_{\rm eff}(x)
=
m_0^2\Omega^2
+(1-6\xi)\Omega^{-1}\nabla^2\Omega.
\label{eq:Veff_def}
\end{equation}
Then the geometry-to-spectrum map is
\begin{equation}
\boxed{
[-\nabla^2+V_{\rm eff}(x)]\psi_N=\omega_N^2\psi_N.
}
\label{eq:schrodinger_operator}
\end{equation}
Together with Eq.~\eqref{eq:Omega_solution}, this gives the PSLT chain
\begin{equation}
\Omega
\longrightarrow
V_{\rm eff}
\longrightarrow
\{\omega_N,\psi_N\}.
\label{eq:geometry_chain}
\end{equation}

\subsection{\texorpdfstring{Two-center Schur-complement kinetics \(\Gamma_N\)}{Two-center Schur-complement kinetics GammaN}}

\begin{equation}
\cC_N=\Span\{|N,+\rangle,\ |N,-\rangle\}.
\end{equation}
Let \(P_N\) project onto \(\cC_N\), \(Q_N=1-P_N\), and let \(\mathcal K_N\) be the Euclidean fluctuation operator obtained from the parent action.
The Feshbach--Schur effective operator is
\begin{equation}
M_N
=
P_N\mathcal K_NP_N
-P_N\mathcal K_NQ_N
(Q_N\mathcal K_NQ_N)^{-1}
Q_N\mathcal K_NP_N.
\label{eq:schur_M}
\end{equation}
In the two-center basis it has the form
\begin{equation}
M_N
=
\begin{pmatrix}
\gamma_{N,+} &
\chi_N\sqrt{\gamma_{N,+}\gamma_{N,-}}\\
\chi_N\sqrt{\gamma_{N,+}\gamma_{N,-}}&
\gamma_{N,-}
\end{pmatrix}.
\label{eq:rank2_matrix}
\end{equation}
The diagonal channel factors are WKB-suppressed:
\begin{equation}
\gamma_{N,\pm}
=
A_{N,\pm}\exp[-2S_{N,\pm}^{\rm WKB}],
\qquad
S_{N,\pm}^{\rm WKB}
=
\int_{\mathcal B_\pm}\dd s\,
\sqrt{V_{\rm eff}(s)-\omega_N^2}.
\label{eq:wkb_gamma}
\end{equation}
The formation rate is
\begin{equation}
\boxed{
\Gamma_N(D,\eta)
=
\max\{0,\lambda_+(M_N(D,\eta))\},
}
\label{eq:Gamma_def}
\end{equation}
where
\begin{equation}
\lambda_+(M_N)
=
\frac{\gamma_{N,+}+\gamma_{N,-}}{2}
+\frac12
\sqrt{
(\gamma_{N,+}-\gamma_{N,-})^2
+4\chi_N^2\gamma_{N,+}\gamma_{N,-}
}.
\label{eq:lambda_plus}
\end{equation}
The rank two structure is therefore a consequence of the two-center collective-coordinate reduction, not a phenomenological closure assumption.

\subsection{\texorpdfstring{Weyl-normalized visibility \(B_N\)}{Weyl-normalized visibility BN}}

\begin{equation}
\Psi=\Omega^{-3/2}\psi,
\qquad
H=\Omega^{-1}\phi.

The Yukawa density obeys
\begin{equation}
\sqrt{-g}\,\bar\Psi H\Psi
\propto
\Omega^4\cdot\Omega^{-3/2}\cdot\Omega^{-1}\cdot\Omega^{-3/2}
=
\Omega^0.
\label{eq:weyl_cancel}
\end{equation}
Thus the minimal conformal baseline has no additional \(\Omega\)-prefactor in the overlap kernel.
For a visible-sector operator \(\mathcal O_{\rm vis}\), define
\begin{equation}
\boxed{
B_N
=
\frac{\langle u_N|\mathcal O_{\rm vis}^\dagger\mathcal O_{\rm vis}|u_N\rangle}
{\sum_{I=1}^{3}
\langle u_I|\mathcal O_{\rm vis}^\dagger\mathcal O_{\rm vis}|u_I\rangle}
}
\label{eq:BN_def}
\end{equation}
or, in Yukawa-overlap form,
\begin{equation}
y_N^{\rm eff}(D)
=
y_0\int 2\pi\rho\,\dd\rho\,\dd z\,
u_N(\rho,z;D)f_L(\rho,z)f_R(\rho,z)W_{\rm frame},
\end{equation}
\begin{equation}
B_N
=
\frac{|y_N^{\rm eff}|^2}
{\sum_{I=1}^{3}|y_I^{\rm eff}|^2}.
\label{eq:BN_yukawa}
\end{equation}
Thus \(B_N\) is a normalized overlap of the same three topological zero modes with the visible sector.

\section{Formation Probability}
\label{sec:formation_probability}

For each chiral layer \(N\), take the effective scalar formation law
\begin{equation}
\frac{\dd p_N}{\dd t}
=
\Gamma_N(1-p_N),
\qquad
p_N(0)=0.
\end{equation}
The solution is
\begin{equation}
p_N(t)=1-\ee^{-\Gamma_Nt}.
\end{equation}
If the \(N\)-th layer has degeneracy \(g_N\) and visible weight \(B_N\), its unnormalized observed formation weight is
\begin{equation}
W_N(t)
=
B_Ng_N\left(1-\ee^{-\Gamma_Nt}\right).
\end{equation}
Conditioning on the event that a chiral layer is observed gives
\begin{equation}
\boxed{
P_N(t;D,\eta)
=
\frac{
B_Ng_N\left(1-\ee^{-\Gamma_N(D,\eta)t}\right)
}{
\sum_{K=1}^{3}
B_Kg_K\left(1-\ee^{-\Gamma_K(D,\eta)t}\right)
}.
}
